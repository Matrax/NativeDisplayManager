312 /*
313 * This methods add a key pressed in the events struct, it also removed the same key from the keys released.
314 * @param virtual_key The virtual key to add.
315 */
316 void AddKeyPressed(int virtual_key)
317 {
318 bool already_pressed = false;
319
320 // Check if the key is already in the array
321 for (size_t i = 0; i < 32 && already_pressed == false; i++)
322 {
323 if (m_events.m_keys_down[i] == virtual_key)
324 already_pressed = true;
325 }
326
327 // If not, we add the key and remove from the keys up array
328 if (already_pressed == false)
329 {
330 for (size_t i = 0; i < 32; i++)
331 {
332 if (m_events.m_keys_down[i] == 0)
333 {
334 m_events.m_keys_down[i] = virtual_key;
335 break;
336 }
337 }
338
339 for (size_t i = 0; i < 32; i++)
340 {
341 if (m_events.m_keys_up[i] == virtual_key)
342 {
343 m_events.m_keys_up[i] = 0;
344 break;
345 }
346 }
347 }
